CREATE TABLE produto(
	codigo char(8),
	nome varchar(15),
	preço real,
	data_de_fabricação date,
	descrição varchar(30),
	especificação varchar (15),
	data_de_validade date,
	PRIMARY KEY (codigo),
	FOREIGN KEY(categoria_produto), REFERENCES categoria(codigo)
	FOREIGN KEY(fornecido), REFERENCES fornecedor(codigo)
);

CREATE TABLE foto_produto(
	foto varchar(max),
	PRIMARY KEY (codigo_produto_foto)
	FOREIGN KEY (codigo_produto_foto), REFERENCES produto(codigo)
);

CREATE TABLE estocado(
	quantidade int,
	PRIMARY KEY (codigo_produto_estocado),
	FOREIGN KEY (codigo_produto_estocado), REFERENCES produto(codigo)
);

CREATE TABLE avalia(
	nota int,
	descricao varchar(30),
	PRIMARY KEY (codigo_produto_avaliacao, codigo_compra_avaliacao),
	FOREIGN KEY (codigo_produto_avaliacao), REFERENCES produto(codigo),
	FOREIGN KEY (codigo_compra_avaliacao), REFERENCES ordem_de_compra(codigo),
);

CREATE TABLE possui(
	quantidade int,
	valor real,
	PRIMARY KEY (codigo_produto_possui, codigo_compra_possui),
	FOREIGN KEY (codigo_produto_possui), REFERENCES produto(codigo),
	FOREIGN KEY (codigo_compra_possui), REFERENCES ordem_de_compra(codigo),
);

CREATE TABLE tem(
	quantidade int,
	PRIMARY KEY (codigo_produto_tem, codigo_carrinho_tem),
	FOREIGN KEY (codigo_produto_tem), REFERENCES produto(codigo),
	FOREIGN KEY (codigo_carrinho_tem), REFERENCES carrinho(codigo)
);
